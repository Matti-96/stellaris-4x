	resources = {
		category = artifact_actions
		cost = {
			minor_artifacts = 50
		}
	}

	potential = {
		always = no # Europe: disabled action because minor artifacts can be bought and sold on the market
		#NAND = {
		#	is_gestalt = yes

		#}
	}

	effect = {

		hidden_effect = {
			add_modifier = {
				modifier = "artifact_sell_cooldown"
				days = 180
			}
		}
	}
	allow = {
		NOT = { has_modifier = artifact_sell_cooldown }

			limit = { is_gestalt = yes }

				fail_text = "GALACTIC_MARKET_NOT_FOUNDED"

			}
		}

			limit = { is_gestalt = yes }
			custom_tooltip = {
				fail_text = "GALACTIC_MARKET_ACCESS"

			}
		}
